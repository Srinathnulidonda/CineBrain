// Simple in-memory cache for preloading
const __cache = new Map(); // key -> { data, expiresAt, promise }
const defaultTTL = 60_000; // 60s

const now = () => Date.now();
const isFresh = (entry) => entry && entry.expiresAt && entry.expiresAt > now();

// Generic real API fetcher with preloading and SWR-like caching
export const fetchRealData = async (
  urlOrEndpoint, 
  {
    method = 'GET',
    params,
    body,
    token = getAuthToken(),
    headers = {},
    ttl = defaultTTL,
    prefetch = false,
    transform, // optional (data) => any
    signal,
  } = {}
) => {
  // Allow passing endpoint path (e.g., '/recommendations/trending') or a full URL
  const isFullUrl = /^https?:\/\//i.test(urlOrEndpoint);
  const base = isFullUrl ? '' : API_ENDPOINTS.BASE_URL;
  const qs = params && Object.keys(params).length ? `?${new URLSearchParams(params).toString()}` : '';
  const url = `${base}${urlOrEndpoint}${qs}`;

  // If cached and fresh, return immediately
  const cached = __cache.get(url);
  if (!prefetch && isFresh(cached) && cached.data !== undefined) {
    return cached.data;
  }
  if (!prefetch && cached?.promise) {
    // Share in-flight request
    const data = await cached.promise;
    return transform ? transform(data) : data;
  }

  const controller = new AbortController();
  const combinedSignal = signal
    ? new AbortController()
    : null;

  if (combinedSignal) {
    // Link signals if provided
    const onAbort = () => controller.abort();
    signal.addEventListener('abort', onAbort, { once: true });
  }

  const fetchPromise = fetch(url, {
    method,
    headers: {
      'Content-Type': 'application/json',
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
      ...headers,
    },
    body: body ? JSON.stringify(body) : undefined,
    signal: controller.signal,
  }).then(async (res) => {
    if (!res.ok) {
      const text = await res.text().catch(() => '');
      const err = new Error(`HTTP ${res.status} ${res.statusText}: ${text || url}`);
      err.status = res.status;
      throw err;
    }
    return res.json();
  });

  // Store in cache immediately to share in-flight promise
  __cache.set(url, {
    promise: fetchPromise,
    data: cached?.data,
    expiresAt: now() + ttl,
  });

  try {
    const data = await fetchPromise;
    const finalData = transform ? transform(data) : data;

    __cache.set(url, {
      promise: null,
      data: finalData,
      expiresAt: now() + ttl,
    });

    return prefetch ? undefined : finalData;
  } catch (e) {
    // On error, clear promise but keep any stale data if available
    __cache.set(url, {
      promise: null,
      data: cached?.data,
      expiresAt: now() + ttl / 4,
    });
    if (prefetch) return undefined;
    throw e;
  }
};
